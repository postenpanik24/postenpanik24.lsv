    script
         Toggle Admin Panel (nur für Demo-Zwecke)
        document.addEventListener('DOMContentLoaded', function() {
             Admin-Bereich anzeigen wenn URL mit admin endet
            if (window.location.href.includes('admin')) {
                document.getElementById('admin-section').style.display = 'block';
            }
            
             Event-Listener für Login-Funktion (Demo)
            const authButtons = document.querySelector('.auth-buttons');
            const loginButton = authButtons.querySelector('afirst-child');
            
            loginButton.addEventListener('click', function(e) {
                e.preventDefault();
                const isAdmin = confirm('Als Admin anmelden');
                if (isAdmin) {
                    document.getElementById('admin-section').style.display = 'block';
                    loginButton.textContent = 'Abmelden';
                    alert('Du bist jetzt als Admin angemeldet!');
                }
            });
            
             Demo-Funktion für Kommentare
            const commentForm = document.querySelector('.comment-form');
            if (commentForm) {
                commentForm.addEventListener('submit', function(e) {
                    e.preventDefault();
                    const commentText = document.getElementById('comment').value;
                    if (commentText.trim() !== '') {
                         Neuen Kommentar erstellen
                        const commentSection = document.querySelector('.comments-section');
                        const newComment = document.createElement('div');
                        newComment.className = 'comment';
                        
                        const now = new Date();
                        const dateStr = now.getDate() + '. ' + 
                                       ['Januar', 'Februar', 'März', 'April', 'Mai', 'Juni', 
                                        'Juli', 'August', 'September', 'Oktober', 'November', 'Dezember'][now.getMonth()] + 
                                       ' ' + now.getFullYear() + ', ' +
                                       now.getHours().toString().padStart(2, '0') + '' + 
                                       now.getMinutes().toString().padStart(2, '0');
                        
                        newComment.innerHTML = `
                            div class=comment-header
                                span class=comment-authorBenutzerspan
                                span class=comment-date${dateStr}span
                            div
                            p class=comment-body${commentText}p
                        `;
                        
                         Kommentar vor das Formular einfügen
                        commentSection.insertBefore(newComment, commentForm);
                        
                         Formular zurücksetzen
                        document.getElementById('comment').value = '';
                        
                        alert('Kommentar erfolgreich hinzugefügt!');
                    }
                });
            }
            
             Demo-Funktion für Admin-Panel
            const adminForm = document.querySelector('.admin-form');
            if (adminForm) {
                adminForm.addEventListener('submit', function(e) {
                    e.preventDefault();
                    const title = document.getElementById('article-title').value;
                    const category = document.getElementById('article-category').value;
                    
                    if (title.trim() !== '') {
                        alert(`Artikel ${title} in der Kategorie ${category} wurde veröffentlicht!`);
                    }
                });
            }
        });
